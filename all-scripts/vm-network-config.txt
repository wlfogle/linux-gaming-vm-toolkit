# Network configuration for Proxmox VM to get IP 192.168.0.65
# Copy this to /etc/network/interfaces in the VM

auto lo
iface lo inet loopback

iface ens3 inet manual

auto vmbr0
iface vmbr0 inet static
        address 192.168.0.65/24
        gateway 192.168.0.1
        bridge-ports ens3
        bridge-stp off
        bridge-fd 0
